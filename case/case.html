# 8.3 下载Schema

在Schema 管理界面 点击下载按钮，下载"chatdemo"对应的graphql标准的shema定义，下载后文件内容如下：

schema {
  query: Querys
  mutation: Mutations
  subscription: Subscriptions
}

type Mutations {
  Createchat(conflict: String = "replace", objects: [chatInput]): insert_return
  CreatechatNoReply(conflict: String = "replace", objects: [chatInput]): Boolean
  Deletechat(where: chatWhereFilter): delete_return
  DeletechatNoReply(where: chatWhereFilter): Boolean
  Updatechat(object: chatUpdateInput, where: chatWhereFilter): update_return
  updatechatNoReply(object: chatUpdateInput, where: chatWhereFilter): Boolean
}

type Querys {
  chatbyid(id: ID!): chatOutput
  chats(first: Int = -1, last: Int = -1, limit: Int = 2147483647, orderby: String, skip: Int = 0, where: chatWhereFilter): chatwhere_result
}

type Subscriptions {
  chatSubscription(mutationtypes: [String], where: chatWhereFilter): chatSubscriptionResult
}

type chatOutput {
  chatroom: String
  createdat: DateTime @createdat
  from: String
  id: ID
  lastupdate: DateTime @lastupdate
  message: String
  receivedtime: String
  sendtime: String
}

type chatSubscriptionResult {
  _typename: String
  mutationtype: String
  node: chatOutput
}

type chatwhere_result {
  result: [chatOutput]
}

type delete_return {
  affectedrow: Int
}

type insert_return {
  affectedrow: Int
}

type update_return {
  affectedrow: Int
}

input IDFilter {
  eq: String
  ge: String
  gt: String
  le: String
  lt: String
  match: String
}

input StringFilter {
  eq: String
  ge: String
  gt: String
  le: String
  lt: String
  match: String
}

input chatFieldFilter {
  chatroom: StringFilter
  from: StringFilter
  id: IDFilter
  message: StringFilter
}

input chatInput {
  chatroom: String!
  from: String!
  id: ID
  message: String!
}

input chatUpdateInput {
  chatroom: String
  from: String
  id: ID
  message: String
}

input chatWhereFilter {
  and: chatWhereFilter
  col: chatFieldFilter
  field_contains: [String]
  not: chatWhereFilter
  or: chatWhereFilter
}

# 8.4 创建React项目

安装create-react-app：

npm install -g create-react-app

创建React项目：

create-react-app graphql-chat

更新package.json为如下内容：

  "dependencies": {
    "react": "^16.8.3",
    "react-dom": "^16.8.3",
    "react-scripts": "2.1.5",
    "apollo-link-ws": "^1.0.4",
    "graphql": "^0.12.3",
    "graphql-tag": "^2.6.1",
    "react-apollo": "^2.0.4",
    "subscriptions-transport-ws": "^0.9.4"
 }

运行npm install：

npm install

在public/index.html的head标签中添加如下两行css加载信息：

<link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/normalize/8.0.0/normalize.min.css" />
<link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/skeleton/2.0.4/skeleton.min.css" />

更新src/index.js：

import { ApolloProvider } from 'react-apollo';
import { ApolloClient } from 'apollo-client';
import { HttpLink } from 'apollo-link-http';
import { InMemoryCache } from 'apollo-cache-inmemory';
import { ApolloLink, split } from 'apollo-client-preset'
import { WebSocketLink } from 'apollo-link-ws'
import { getMainDefinition } from 'apollo-utilities'

在import部分之后，配置websocket连接的信息，如下：

const wsLink = new WebSocketLink({
  uri: '[Subscriptions API URI]',
  options: {
    reconnect: true
  }
})

在websocket的配置后面新增http的配置：

const httpLink = new HttpLink({ uri: '[SIMPLE API URI]' })

到此，我们有两个连接配置，我们需要根据访问请求的不同，决定使用哪个连接：

const link = split(
  ({ query }) => {
    const { kind, operation } = getMainDefinition(query)
    return kind === 'OperationDefinition' && operation === 'subscription'
  },
  wsLink,
  httpLink,
)

创建Apollo Client：

const client = new ApolloClient({
  link,
  cache: new InMemoryCache()
})

用AppProvider提供client：

ReactDOM.render(
  <ApolloProvider client={client}>
    <App />
  </ApolloProvider>,
  document.getElementById('root')
);

const ALL_CHATS_QUERY = gql`
  query AllChatsQuery {
    Chat_Where {
      id
      createdAt
      from
      content
    }
  }
`;

export default graphql(ALL_CHATS_QUERY, { name: 'allChatsQuery' })(App);

以上代码主要完成以下三方面工作：

  • 首先引入了graphql和gql，这两个类库会帮助构建和解析查询语句。
  • 在组件定义的最后，用gql标签创建了查询。
  • 用graphql将查询结果的值付给App组件的props。

上述代码完成了聊天记录的查询和展示，下面是聊天消息的发送。类似于查询，我们创建一个mutation

import { graphql, compose } from 'react-apollo';

    // App component ...

    const CREATE_CHAT_MUTATION = gql`
      mutation CreateChatMutation($content: String!, $from: String!，$createdAt DateTime) {
        insert_Chat(objects:[{$content, from: $from,createdAt:$createdAt}]) {
          id
          createdAt
          from
          content
        }
      }
    `;

    export default compose(
      graphql(ALL_CHATS_QUERY, { name: 'allChatsQuery' }),
      graphql(CREATE_CHAT_MUTATION, { name: 'createChatMutation' })
    )(App);

mutation在形式上类似于query，但它需要两个参数from和content。我们需要同时用mutation和query来封装App组件。我们在render方法，用一个input元素来收集聊天信息。

render() {
   const allChats = this.props.allChatsQuery.allChats || [];
       return (
          <div className="">
            <div className="container">
              <h2>Chats</h2>
              {allChats.map(message => (
                <Chatbox key={message.id} message={message} />
              ))}

              {/* Message content input */}
              <input
                value={this.state.content}
                onChange={e => this.setState({ content: e.target.value })}
                type="text"
                placeholder="Start typing"
                onKeyPress={this._createChat}
              />
            </div>
          </div>
        );
      }
    }

当用户按下enter键时完成用户的输入，在App.js中增加_createChat方法：

_createChat = async e => {
     if (e.key === 'Enter') {
       const { content, from } = this.state;
        await this.props.createChatMutation({
          variables: { content, from }
        });
        this.setState({ content: '' });
      }
    };

接下来在App.js添加实时订阅：

_subscribeToNewChats = () => {
      this.props.allChatsQuery.subscribeToMore({
          document: gql`
            subscription {
              sub_Chat {
                new_val {
                  id
                  from
                  content
                  createdAt
                }
              }
            }
          `,
          updateQuery: (previous, { subscriptionData }) => {
            const newChatLinks = [
              ...previous.allChats,
              subscriptionData.data.sub_Chat.new_val
            ];
            const result = {
              ...previous,
              allChats: newChatLinks
            };
            return result;
          }
        });
      };
The allChatsQuery exposes a

allChatsQuery暴露了一个subscribeToMore方法，一旦这个方法被调用，就打开了一个实时订阅的通道，这个方法每获得一个对象，都会调用updateQuery方法。

我们在componentDidMount触发subscribeToMore

this._subscribeToNewChats();
